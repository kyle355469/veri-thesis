\begin{denotation}[3cm]

\item[LLM]{
  Large Language Model（大型語言模型）
}

\item[RTL]{
  Register-Transfer Level（暫存器傳輸級）
}

\item[HDL]{
  Hardware Description Language（硬體描述語言）
}

\item[IP]{
  Intellectual Property core（矽智財；可重用之硬體設計區塊）
}

\item[DUT]{
  Device Under Test（受測設計；測試平台所例化之目標模組）
}

\item[FSM]{
  Finite-State Machine（有限狀態機）
}

\item[CDC]{
  Clock-Domain Crossing（跨時脈域）
}

\item[RAG]{
  Retrieval-Augmented Generation（檢索增強生成）
}

\item[AES]{
  Advanced Encryption Standard（進階加密標準）
}

\item[SDC]{
  SD-Card Controller（SD 卡控制器；本文之任務家族名稱）
}

\item[E203]{
  Nuclei HummingBirdv2 E203 開源 RISC-V 處理器核心
}

\item[vLLM]{
  高吞吐量 LLM 推論服務框架（PagedAttention）
}

\item[pass@$k$]{
  $k$ 次取樣中至少一次通過之無偏估計通過率
}

\item[syntax rate]{
  通過基準測試編譯（無任何錯誤與未豁免警告）之樣本比例
}

\item[pass rate]{
  同時通過編譯與測試平台功能比對之樣本比例
}

\item[$s/f$]{
  修復預算表記：每樣本至多 $s$ 次語法修復與 $f$ 次功能修復
}

\item[MODDUP]{
  Verilator 錯誤類別：重複宣告環境已提供之模組
}

\item[PINNOTFOUND]{
  Verilator 錯誤類別：例化埠不存在（違反介面契約）
}

\end{denotation}
